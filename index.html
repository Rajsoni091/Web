Governor Limit Benefit: Batch Apex resets limits for every chunk of 200. You can query 50M records total without hitting SOQL limits.

Batch Apex Skeleton

public class InactiveCustomerBatch implements Database.Batchable<SObject> {

    public Database.QueryLocator start(Database.BatchableContext bc) {
        Date cutoff = Date.today().addMonths(-12);
        return Database.getQueryLocator(
            'SELECT Id, Last_Order_Date__c, Is_Active_Buyer__c ' +
            'FROM Customer__c WHERE Last_Order_Date__c < :cutoff'
        );
    }

    public void execute(Database.BatchableContext bc, List<Customer__c> scope) {
        for (Customer__c c : scope) {
            c.Is_Active_Buyer__c = false;
            c.Inactive_Flagged_On__c = Date.today();
        }
        update scope;
    }

    public void finish(Database.BatchableContext bc) {
        // Notify admin, chain next job, etc.
    }
}
// Execute:  Database.executeBatch(new InactiveCustomerBatch(), 200);

public class InventoryUpdateJob implements Queueable, Database.AllowsCallouts {
    private Id orderId;
    public InventoryUpdateJob(Id orderId) { this.orderId = orderId; }

    public void execute(QueueableContext ctx) {
        // 1. Update inventory
        update [SELECT Id, Stock__c FROM Product__c WHERE Id IN :getProductIds()];

        // 2. Enqueue the next job in the chain
        if (!Test.isRunningTest()) {
            System.enqueueJob(new ExternalNotifyJob(orderId));
        }
    }
}
Why chain instead of @future? Queueables accept Apex objects as params, support chaining, and give you a jobId to monitor — @future can't do any of that.
